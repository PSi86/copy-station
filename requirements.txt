# Runtime dependencies of Copy_Station
#
# Hardware dependencies (only needed on the Cubie / Linux) are installed via
# system packages, NOT via pip:
#   apt install rsync python3-pyudev python3-libgpiod python3-spidev
#
# The optional e-paper backend renders with Pillow; on the device it comes from
# apt (python3-pil + fonts-dejavu-core, installed by scripts/install.sh).
#
# This list only covers the pure-Python dependencies that also run on the
# Windows dev machine.

PyYAML>=5.3

# Web interface (only needed when web.enabled = true):
fastapi>=0.100
uvicorn>=0.20

# The optional web features need NO extra pip packages -- they rely on system
# tooling installed by scripts/install.sh:
#   * File browser + WiFi AP: mount/umount and NetworkManager (`nmcli`).
#   * Video transcoding:      `ffmpeg` (also provides `ffprobe`).
# NetworkManager is the default network stack on Raspberry Pi OS Bookworm and
# current Radxa images; it is not auto-installed to avoid clashing with an
# existing dhcpcd/networkd setup.

# Development/tests only:
#   pytest>=6.0
#   httpx>=0.24   (FastAPI TestClient)
#   Pillow>=9.0   (e-paper layout/backend tests)
